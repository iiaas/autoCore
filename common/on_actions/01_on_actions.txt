on_actions = {
	on_army_leader_daily = {
		effect = {
			if = {
				limit = { FROM = { is_ai = no } }
				supply_units = 24
			}
		}
	}

	on_daily = {
		effect = {
			if = {
				limit = { is_ai = no }
				auto_research = yes
				auto_build = yes
				if = {
					limit = { has_idea = autocore_army_no_fuel }
					modify_timed_idea = { idea = autocore_army_no_fuel days = 1 }
					auto_production_naval = yes
				}
				else = {
					add_timed_idea = { idea = autocore_army_no_fuel days = 2 }
					add_daily_mastery = { amount = 1 days = 2000 }
					auto_mio_complete_trait = yes
					auto_design = yes
					auto_focus_shine = yes
					auto_production = yes
				}
			}
		}
	}

	on_monthly = {
		effect = {
			if = {
				limit = { is_ai = no }
				auto_production = yes
				auto_focus_complete = yes
			}
		}
	}
}
