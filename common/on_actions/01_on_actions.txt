on_actions = {
	on_daily = {
		effect = {
			if = {
				limit = { is_ai = no }
				auto_research = yes
				auto_build = yes
				auto_focus_shine = yes
				# 供决议倒计时使用：剩余天数 = 总天数 - 已推进天数（决议任务在暂停后重新激活时
				# 以此为 days_mission_timeout，从正确的剩余天数继续倒数）
				set_variable = { autocore_focus_remaining = autocore_focus_total }
				subtract_from_variable = { autocore_focus_remaining = autocore_focus_days }
				auto_production_naval = yes
				auto_mio_complete_trait = yes
				# 设计/MIO 只执行一次（永久 flag 复刻原 autocore_army_no_fuel idea 的
				# "加上就永不过期"节奏）：auto_design 重复执行会不断重复创建变体/师模板
				if = {
					limit = { NOT = { has_country_flag = auto_design_done } }
					auto_design = yes
					set_country_flag = auto_design_done
				}
			}
		}
	}
}
