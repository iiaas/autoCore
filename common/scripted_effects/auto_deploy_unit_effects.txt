# 在首都的胜利点省份生成守备师。
# 脚本运行时读不到胜利点数据（引擎只把 GetCapitalVictoryPointName 开放给本地化），
# 故仅对开局主要国家 + 中国的首都州硬编码 VP 省份（数据核对自 history/states/*.txt）；
# 其他首都走 else 不指定 prioritize_location，由引擎在首都州自选落点。
auto_deploy_unit = {
	if = {
		limit = {
			has_template = "Infanterie-Division"
			has_equipment = { infantry_equipment > 2500 }
			divisions_in_state = { state = ROOT.capital size < 24 }
		}
		if = {
			# 德国·柏林
			limit = { original_tag = GER }
			capital_scope = { create_unit = { division = "division_template = \"Infanterie-Division\" start_experience_factor = 0.5 start_equipment_factor = 0.1" owner = ROOT count = 24 prioritize_location = 6521 } }
		}
		else_if = {
			# 苏联·莫斯科
			limit = { original_tag = SOV }
			capital_scope = { create_unit = { division = "division_template = \"Infanterie-Division\" start_experience_factor = 0.5 start_equipment_factor = 0.1" owner = ROOT count = 24 prioritize_location = 6380 } }
		}
		else_if = {
			# 英国·伦敦
			limit = { original_tag = ENG }
			capital_scope = { create_unit = { division = "division_template = \"Infanterie-Division\" start_experience_factor = 0.5 start_equipment_factor = 0.1" owner = ROOT count = 24 prioritize_location = 6103 } }
		}
		else_if = {
			# 法国·巴黎
			limit = { original_tag = FRA }
			capital_scope = { create_unit = { division = "division_template = \"Infanterie-Division\" start_experience_factor = 0.5 start_equipment_factor = 0.1" owner = ROOT count = 24 prioritize_location = 11506 } }
		}
		else_if = {
			# 意大利·罗马
			limit = { original_tag = ITA }
			capital_scope = { create_unit = { division = "division_template = \"Infanterie-Division\" start_experience_factor = 0.5 start_equipment_factor = 0.1" owner = ROOT count = 24 prioritize_location = 9904 } }
		}
		else_if = {
			# 美国·华盛顿
			limit = { original_tag = USA }
			capital_scope = { create_unit = { division = "division_template = \"Infanterie-Division\" start_experience_factor = 0.5 start_equipment_factor = 0.1" owner = ROOT count = 24 prioritize_location = 3957 } }
		}
		else_if = {
			# 日本·东京
			limit = { original_tag = JAP }
			capital_scope = { create_unit = { division = "division_template = \"Infanterie-Division\" start_experience_factor = 0.5 start_equipment_factor = 0.1" owner = ROOT count = 24 prioritize_location = 1182 } }
		}
		else_if = {
			# 国民党中国·南京
			limit = { original_tag = CHI }
			capital_scope = { create_unit = { division = "division_template = \"Infanterie-Division\" start_experience_factor = 0.5 start_equipment_factor = 0.1" owner = ROOT count = 24 prioritize_location = 11913 } }
		}
		else = {
			# 其他首都：不指定落点，引擎在首都州自选
			capital_scope = { create_unit = { division = "division_template = \"Infanterie-Division\" start_experience_factor = 0.5 start_equipment_factor = 0.1" owner = ROOT count = 24 } }
		}
	}
}
